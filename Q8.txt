CREATE PROCEDURE check_result(IN marks INT)
BEGIN
    IF marks < 0 OR marks > 100 THEN
        SELECT 'Invalid marks! Please enter marks between 0 and 100.' AS Result;
    ELSEIF marks >= 75 THEN
        SELECT 'Distinction' AS Result;
    ELSEIF marks >= 60 THEN
        SELECT 'First Class' AS Result;
    ELSEIF marks >= 50 THEN
        SELECT 'Second Class' AS Result;
    ELSE
        SELECT 'Fail' AS Result;
    END IF;
END;
//

-- Reset the delimiter back to semicolon
DELIMITER ;
